-- Create the departments table
CREATE TABLE departments (
    department_id INT PRIMARY KEY,
    department_name VARCHAR(50)
);

-- Create the employees table
CREATE TABLE employees (
    employee_id INT PRIMARY KEY,
    employee_name VARCHAR(50),
    department_id INT,
    salary DECIMAL(10, 2),
    FOREIGN KEY (department_id) REFERENCES departments(department_id)
);

List all employees along with their department names (INNER JOIN).


SELECT e.employee_name, e.salary, d.department_name
FROM employees e
JOIN departments d ON e.department_id = d.department_id;



List all employees, including those not assigned to any department (LEFT JOIN).


SELECT e.employee_name, e.salary, d.department_name
FROM employees e
LEFT JOIN departments d ON e.department_id = d.department_id;

List all departments, including those without employees (RIGHT JOIN).

SELECT e.employee_name, e.salary, d.department_name
FROM employees e
RIGHT JOIN departments d ON e.department_id = d.department_id;

List all employees and departments, including those that do not match (FULL JOIN using UNION).



SELECT e.employee_name, e.salary, d.department_name
FROM employees e
LEFT JOIN departments d ON e.department_id = d.department_id
UNION
SELECT e.employee_name, e.salary, d.department_name
FROM employees e
RIGHT JOIN departments d ON e.department_id = d.department_id;



List all possible pairs of employees and departments (CROSS JOIN).

SELECT e.employee_name, d.department_name
FROM employees e
CROSS JOIN departments d;

List employees with a salary greater than $60,000 with department_name (INNER JOIN)

SELECT e.employee_name, e.salary, d.department_name
FROM employees e
JOIN departments d ON e.department_id = d.department_id
WHERE e.salary > 60000;

List employees who work in the 'Engineering' department (INNER JOIN and WHERE).

SELECT e.employee_name, e.salary,d.department_name
FROM employees e
JOIN departments d ON e.department_id = d.department_id
WHERE d.department_name = 'Engineering';


Find departments that have no employees (LEFT JOIN and WHERE).

SELECT d.department_name
FROM departments d
LEFT JOIN employees e ON d.department_id = e.department_id
WHERE e.employee_id IS NULL;

Find employees and their departments where the employee name starts with 'A' (LEFT JOIN and WHERE)

SELECT e.employee_name, e.salary, d.department_name
FROM employees e
LEFT JOIN departments d ON e.department_id = d.department_id
WHERE e.employee_name LIKE 'A%';
